                )
            }
        })
    }
    finally (cb) {
        return this.then(
            (value) => this.constructor.resolve(cb()).then(() => value),
            (reason) =>
                this.constructor.resolve(cb()).then(() => {
                    throw reason
                })
        )
    }
}
